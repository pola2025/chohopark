    <script>
        // 외부시설 갤러리 전용 데이터
        const outdoorGalleries = {
            outdoor: {
                currentSlide: 0,
                images: [
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/3968dd8bada02.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/862fde0ac110f.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/0105b4b8158b4.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/fd88c15fce28f.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/daf47a0a40e30.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/c9fb99ab14fcd.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/d2be5f2ae3730.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/2b0ecaa3aa257.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/32b2f913ef535.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/200af3dfb72ad.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/c155843b98437.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/e9719d780dccf.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/034180c3244a3.jpg'
                ]
            }
        };
        
        let outdoorCurrentGallery = 'outdoor';
        let outdoorModalIndex = 0;
        let outdoorAutoPlayInterval;
        
        // 외부시설 자동 롤링 시작
        function startOutdoorAutoPlay() {
            outdoorAutoPlayInterval = setInterval(() => {
                changeOutdoorSlide(1, 'outdoor');
            }, 4000); // 4초마다 자동 전환
        }
        
        // 외부시설 자동 롤링 정지
        function stopOutdoorAutoPlay() {
            if (outdoorAutoPlayInterval) {
                clearInterval(outdoorAutoPlayInterval);
            }
        }
        
        // 외부시설 슬라이드 변경
        function changeOutdoorSlide(direction, galleryType) {
            stopOutdoorAutoPlay(); // 수동 조작시 자동 롤링 일시 정지
            
            const gallery = outdoorGalleries[galleryType];
            gallery.currentSlide += direction;
            
            if (gallery.currentSlide < 0) {
                gallery.currentSlide = gallery.images.length - 1;
            } else if (gallery.currentSlide >= gallery.images.length) {
                gallery.currentSlide = 0;
            }
            
            updateOutdoorGallery(galleryType);
            startOutdoorAutoPlay(); // 다시 자동 롤링 시작
        }
        
        // 외부시설 특정 슬라이드로 이동
        function goToOutdoorSlide(index, galleryType) {
            stopOutdoorAutoPlay(); // 수동 조작시 자동 롤링 일시 정지
            outdoorGalleries[galleryType].currentSlide = index;
            updateOutdoorGallery(galleryType);
            startOutdoorAutoPlay(); // 다시 자동 롤링 시작
        }
        
        // 외부시설 갤러리 업데이트
        function updateOutdoorGallery(galleryType) {
            const gallery = outdoorGalleries[galleryType];
            const wrapper = document.getElementById(galleryType + 'GalleryWrapper');
            const transform = -gallery.currentSlide * 100;
            wrapper.style.transform = `translateX(${transform}%)`;
            
            // 인디케이터 업데이트
            const indicators = document.querySelectorAll(`#${galleryType}Indicators .indicator`);
            indicators.forEach((indicator, index) => {
                indicator.classList.toggle('active', index === gallery.currentSlide);
            });
            
            // 썸네일 업데이트 - 현재 슬라이드 기준으로 4개만 표시
            const thumbnails = document.querySelectorAll('#s2025081074d7e8b1c53b7 .thumbnail');
            const currentSlide = gallery.currentSlide;
            
            // 표시할 썸네일 인덱스 계산
            let startIndex = Math.max(0, currentSlide - 1);
            if (startIndex + 4 > gallery.images.length) {
                startIndex = Math.max(0, gallery.images.length - 4);
            }
            
            thumbnails.forEach((thumbnail, index) => {
                const imgElement = thumbnail.querySelector('img');
                const targetIndex = startIndex + index;
                
                if (targetIndex < gallery.images.length) {
                    imgElement.src = gallery.images[targetIndex];
                    thumbnail.onclick = () => goToOutdoorSlide(targetIndex, galleryType);
                    thumbnail.classList.toggle('active', targetIndex === currentSlide);
                }
            });
        }
        
        // 외부시설 모달 열기
        function openOutdoorModal(index, galleryType) {
            outdoorCurrentGallery = galleryType;
            outdoorModalIndex = index;
            const modal = document.getElementById('outdoorModal');
            const modalImage = document.getElementById('modalImage');
            modalImage.src = outdoorGalleries[galleryType].images[index];
            modal.classList.add('active');
            document.body.style.overflow = 'hidden';
        }
        
        // 외부시설 모달 닫기
        function closeOutdoorModal() {
            const modal = document.getElementById('outdoorModal');
            modal.classList.remove('active');
            document.body.style.overflow = '';
        }
        
        // 외부시설 모달 네비게이션
        function outdoorModalNav(direction) {
            const gallery = outdoorGalleries[outdoorCurrentGallery];
            outdoorModalIndex += direction;
            
            if (outdoorModalIndex < 0) {
                outdoorModalIndex = gallery.images.length - 1;
            } else if (outdoorModalIndex >= gallery.images.length) {
                outdoorModalIndex = 0;
            }
            
            const modalImage = document.getElementById('modalImage');
            modalImage.src = gallery.images[outdoorModalIndex];
        }
        
        // ESC 키로 모달 닫기
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape') {
                closeOutdoorModal();
            }
        });
        
        // 모달 배경 클릭으로 닫기
        document.getElementById('outdoorModal').addEventListener('click', (e) => {
            if (e.target === e.currentTarget) {
                closeOutdoorModal();
            }
        });
        
        // 페이지 로드시 자동 롤링 시작
        window.addEventListener('load', () => {
            startOutdoorAutoPlay();
        });
        
        // 페이지 벗어날 때 자동 롤링 정지
        window.addEventListener('beforeunload', () => {
            stopOutdoorAutoPlay();
        });
    </script>
