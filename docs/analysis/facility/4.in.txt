        let indoorCurrentGallery = 'indoor';
        let indoorModalIndex = 0;
        let indoorAutoPlayInterval;
        
        // 내부시설 자동 롤링 시작
        function startIndoorAutoPlay() {
            indoorAutoPlayInterval = setInterval(() => {
                changeIndoorSlide(1, 'indoor');
            }, 4000); // 4초마다 자동 전환
        }
        
        // 내부시설 자동 롤링 정지
        function stopIndoorAutoPlay() {
            if (indoorAutoPlayInterval) {
                clearInterval(indoorAutoPlayInterval);
            }
        }
        
        // 내부시설 슬라이드 변경
        function changeIndoorSlide(direction, galleryType) {
            stopIndoorAutoPlay();
            
            const gallery = indoorGalleries[galleryType];
            gallery.currentSlide += direction;
            
            if (gallery.currentSlide < 0) {
                gallery.currentSlide = gallery.images.length - 1;
            } else if (gallery.currentSlide >= gallery.images.length) {
                gallery.currentSlide = 0;
            }
            
            updateIndoorGallery(galleryType);
            startIndoorAutoPlay();
        }
        
        // 내부시설 특정 슬라이드로 이동
        function goToIndoorSlide(index, galleryType) {
            stopIndoorAutoPlay();
            indoorGalleries[galleryType].currentSlide = index;
            updateIndoorGallery(galleryType);
            startIndoorAutoPlay();
        }
        
        // 내부시설 갤러리 업데이트
        function updateIndoorGallery(galleryType) {
            const gallery = indoorGalleries[galleryType];
            const wrapper = document.getElementById(galleryType + 'GalleryWrapper');
            const transform = -gallery.currentSlide * 100;
            wrapper.style.transform = `translateX(${transform}%)`;
            
            // 인디케이터 업데이트
            const indicators = document.querySelectorAll(`#${galleryType}Indicators .indicator`);
            indicators.forEach((indicator, index) => {
                indicator.classList.toggle('active', index === gallery.currentSlide);
            });
            
            // 썸네일 업데이트
            const thumbnails = document.querySelectorAll('#s20250810823465814c970 .thumbnail');
            const currentSlide = gallery.currentSlide;
            
            let startIndex = Math.max(0, currentSlide - 1);
            if (startIndex + 4 > gallery.images.length) {
                startIndex = Math.max(0, gallery.images.length - 4);
            }
            
            thumbnails.forEach((thumbnail, index) => {
                const imgElement = thumbnail.querySelector('img');
                const targetIndex = startIndex + index;
                
                if (targetIndex < gallery.images.length) {
                    imgElement.src = gallery.images[targetIndex];
                    thumbnail.onclick = () => goToIndoorSlide(targetIndex, galleryType);
                    thumbnail.classList.toggle('active', targetIndex === currentSlide);
                }
            });
        }
        
        // 내부시설 모달 열기
        function openIndoorModal(index, galleryType) {
            indoorCurrentGallery = galleryType;
            indoorModalIndex = index;
            const modal = document.getElementById('indoorModal');
            const modalImage = document.getElementById('indoorModalImage');
            modalImage.src = indoorGalleries[galleryType].images[index];
            modal.classList.add('active');
            document.body.style.overflow = 'hidden';
        }
        
        // 내부시설 모달 닫기
        function closeIndoorModal() {
            const modal = document.getElementById('indoorModal');
            modal.classList.remove('active');
            document.body.style.overflow = '';
        }
        
        // 내부시설 모달 네비게이션
        function indoorModalNav(direction) {
            const gallery = indoorGalleries[indoorCurrentGallery];
            indoorModalIndex += direction;
            
            if (indoorModalIndex < 0) {
                indoorModalIndex = gallery.images.length - 1;
            } else if (indoorModalIndex >= gallery.images.length) {
                indoorModalIndex = 0;
            }
            
            const modalImage = document.getElementById('indoorModalImage');
            modalImage.src = gallery.images[indoorModalIndex];
        }
        
        // ESC 키로 모달 닫기 (내부시설 전용)
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape' && document.getElementById('indoorModal').classList.contains('active')) {
                closeIndoorModal();
            }
        });
        
        // 모달 배경 클릭으로 닫기
        document.getElementById('indoorModal').addEventListener('click', (e) => {
            if (e.target === e.currentTarget) {
                closeIndoorModal();
            }
        });

            stopIndoorAutoPlay();
