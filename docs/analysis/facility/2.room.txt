    <script>
        // 갤러리 데이터
        const galleries = {
            room: {
                currentSlide: 0,
                images: [
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/bf583107a6070.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/77aa7804dc564.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/ccfe7cc7f3221.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/65c9b0373ddca.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/ef04234d1f310.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/ca3e4a4cbe6f8.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/45e6ea9779d89.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/e0c93756c1287.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/95e1b4a9ac45b.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/b99fe86b0cff3.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/535c3cc43789c.jpg',
                    'https://cdn.imweb.me/upload/S202110280a2b6da12f806/ac713e5b11d95.jpg'
                ]
            }
        };
        
        let currentGallery = 'room';
        let modalIndex = 0;
        let autoPlayInterval;
        
        // 자동 롤링 시작
        function startAutoPlay() {
            autoPlayInterval = setInterval(() => {
                changeSlide(1, 'room');
            }, 4000); // 4초마다 자동 전환
        }
        
        // 자동 롤링 정지
        function stopAutoPlay() {
            if (autoPlayInterval) {
                clearInterval(autoPlayInterval);
            }
        }
        
        // 슬라이드 변경
        function changeSlide(direction, galleryType) {
            stopAutoPlay(); // 수동 조작시 자동 롤링 일시 정지
            
            const gallery = galleries[galleryType];
            gallery.currentSlide += direction;
            
            if (gallery.currentSlide < 0) {
                gallery.currentSlide = gallery.images.length - 1;
            } else if (gallery.currentSlide >= gallery.images.length) {
                gallery.currentSlide = 0;
            }
            
            updateGallery(galleryType);
            startAutoPlay(); // 다시 자동 롤링 시작
        }
        
        // 특정 슬라이드로 이동
        function goToSlide(index, galleryType) {
            stopAutoPlay(); // 수동 조작시 자동 롤링 일시 정지
            galleries[galleryType].currentSlide = index;
            updateGallery(galleryType);
            startAutoPlay(); // 다시 자동 롤링 시작
        }
        
        // 갤러리 업데이트
        function updateGallery(galleryType) {
            const gallery = galleries[galleryType];
            const wrapper = document.getElementById(galleryType + 'GalleryWrapper');
            const transform = -gallery.currentSlide * 100;
            wrapper.style.transform = `translateX(${transform}%)`;
            
            // 인디케이터 업데이트
            const indicators = document.querySelectorAll(`#${galleryType}Indicators .indicator`);
            indicators.forEach((indicator, index) => {
                indicator.classList.toggle('active', index === gallery.currentSlide);
            });
            
            // 썸네일 업데이트 - 현재 슬라이드 기준으로 4개만 표시
            const thumbnails = document.querySelectorAll('.thumbnail');
            const currentSlide = gallery.currentSlide;
            
            // 표시할 썸네일 인덱스 계산
            let startIndex = Math.max(0, currentSlide - 1);
            if (startIndex + 4 > gallery.images.length) {
                startIndex = Math.max(0, gallery.images.length - 4);
            }
            
            thumbnails.forEach((thumbnail, index) => {
                const imgElement = thumbnail.querySelector('img');
                const targetIndex = startIndex + index;
                
                if (targetIndex < gallery.images.length) {
                    imgElement.src = gallery.images[targetIndex];
                    thumbnail.onclick = () => goToSlide(targetIndex, galleryType);
                    thumbnail.classList.toggle('active', targetIndex === currentSlide);
                }
            });
        }
        
        // 모달 열기
        function openModal(index, galleryType) {
            currentGallery = galleryType;
            modalIndex = index;
            const modal = document.getElementById('imageModal');
            const modalImage = document.getElementById('modalImage');
            modalImage.src = galleries[galleryType].images[index];
            modal.classList.add('active');
            document.body.style.overflow = 'hidden';
        }
        
        // 모달 닫기
        function closeModal() {
            const modal = document.getElementById('imageModal');
            modal.classList.remove('active');
            document.body.style.overflow = '';
        }
        
        // 모달 네비게이션
        function modalNav(direction) {
            const gallery = galleries[currentGallery];
            modalIndex += direction;
            
            if (modalIndex < 0) {
                modalIndex = gallery.images.length - 1;
            } else if (modalIndex >= gallery.images.length) {
                modalIndex = 0;
            }
            
            const modalImage = document.getElementById('modalImage');
            modalImage.src = gallery.images[modalIndex];
        }
        
        // ESC 키로 모달 닫기
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape') {
                closeModal();
            }
        });
        
        // 모달 배경 클릭으로 닫기
        document.getElementById('imageModal').addEventListener('click', (e) => {
            if (e.target === e.currentTarget) {
                closeModal();
            }
        });
        
        // 페이지 로드시 자동 롤링 시작
        window.addEventListener('load', () => {
            startAutoPlay();
        });
        
        // 페이지 벗어날 때 자동 롤링 정지
        window.addEventListener('beforeunload', () => {
            stopAutoPlay();
        });
    </script>
